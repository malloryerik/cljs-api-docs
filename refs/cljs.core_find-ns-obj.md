```clj
(defn find-ns-obj [ns]
  (let [munged-ns (munge (str ns))
        segs (.split munged-ns ".")]
    (case *target*
      "nodejs"  (if ^boolean js/COMPILED
                  (js/eval munged-ns)
                  (find-ns-obj* js/global segs))
      "default" (find-ns-obj* js/window segs)
      (throw (js/Error. (str "find-ns-obj not supported for target " *target*))))))
```
