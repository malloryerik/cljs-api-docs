```clj
(deftype SeqIter [^:mutable seq]
  Object
  (hasNext [_] (not (nil? seq)))
  (next [_]
    (let [first (first seq)]
      (set! seq (next seq))
      first))
  (remove [_] (js/Error. "Unsupported operation")))
```
