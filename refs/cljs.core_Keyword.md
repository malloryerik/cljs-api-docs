```clj
(deftype Keyword [k]
  IFn
  (invoke [_ coll]
    (when-not (nil? coll)
      (when (satisfies? ILookup coll)
        (-lookup coll k nil))))
  (invoke [_ coll not-found]
    (if (nil? coll)
      not-found
      (when (satisfies? ILookup coll)
        (-lookup coll k not-found)))))
```
