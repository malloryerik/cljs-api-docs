```
Like test-var, but returns a block for further composition and
later execution.
```


Source code @ [github](https://github.com/clojure/clojurescript/blob/r2850/src/cljs/cljs/test.cljs#L437-L459):

```clj
(defn test-var-block
  [v]
  {:pre [(instance? Var v)]}
  (if-let [t (:test (meta v))]
    [(fn []
       (update-current-env! [:testing-vars] conj v)
       (update-current-env! [:report-counters :test] inc)
       (do-report {:type :begin-test-var :var v})
       (let [{:keys [async-disabled]} (get-current-env)]
         (cond-> (try
                   (t)
                   (catch :default e
                     (do-report
                      {:type :error
                       :message "Uncaught exception, not in assertion."
                       :expected nil
                       :actual e})))
           async-disabled (-> async? not (assert async-disabled)))))
     (fn []
       (do-report {:type :end-test-var :var v})
       (update-current-env! [:testing-vars] rest))]))
```
