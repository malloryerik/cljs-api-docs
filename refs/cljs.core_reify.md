```clj
(defmacro reify [& impls]
  (let [t (gensym "t")
        locals (keys (:locals &env))]
    `(do
       (when (undefined? ~t)
         (deftype ~t [~@locals __meta#]
           cljs.core.IWithMeta
           (~'-with-meta [_# __meta#]
             (new ~t ~@locals __meta#))
           cljs.core.IMeta
           (~'-meta [_#] __meta#)
           ~@impls))
       (new ~t ~@locals nil))))
```
