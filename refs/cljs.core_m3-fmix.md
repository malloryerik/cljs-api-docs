```clj
(defn ^number m3-fmix [h1 len]
  (as-> h1 h1
    (bit-xor h1 len)
    (bit-xor h1 (unsigned-bit-shift-right h1 16))
    (imul h1 0x85ebca6b)
    (bit-xor h1 (unsigned-bit-shift-right h1 13))
    (imul h1 0xc2b2ae35)
    (bit-xor h1 (unsigned-bit-shift-right h1 16))))
```
