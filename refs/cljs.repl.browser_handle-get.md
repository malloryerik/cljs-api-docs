```clj
(defn handle-get [opts conn request]
  (let [path (:path request)]
    (cond
     (.startsWith path "/repl") (send-repl-client-page opts conn request)
     (:serve-static opts) (send-static opts conn request)
     :else (send-404 conn (:path request)))))
```
