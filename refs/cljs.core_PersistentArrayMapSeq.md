  Object
  (toString [coll]
    (pr-str* coll))
  (equiv [this other]

  IMeta
  (-meta [coll] _meta)

  IWithMeta
  (-with-meta [coll new-meta]
    (PersistentArrayMapSeq. arr i new-meta))

  ICounted
  (-count [coll]
    (/ (- (alength arr) i) 2))

  ISeqable
  (-seq [coll] coll)

  ISequential
  IEquiv
  (-equiv [coll other] (equiv-sequential coll other))

  ICollection
  (-conj [coll o]

  IEmptyableCollection
  (-empty [coll] (with-meta (.-EMPTY List) _meta))

  ISeq
